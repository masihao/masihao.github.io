	   & {\it #3}\\
	\end{tabular}\par}

\newcommand\LongestDate{\widthof{7777 (Spring)}}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% DatedItem: A date followed by an item
% USAGE:
%	\DatedItem{DATE}{ITEM}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newcommand\DatedItem[2]{
	\begin{tabular}[t]{@{\hspace{0.125in}}p{\linewidth - \LongestDate - 1.575in}@{\hspace{0.25in}}p{\LongestDate}@{}}
	{\raggedright #1}	&#2  \\
	\end{tabular}\par}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% NumberedItem: A number followed by an item (max 2 digit number)
% USAGE:
%       \NumberedItem{NUM}{ITEM}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newcommand\NumberedItem[2]{
        \begin{tabular}[t]{@{\hspace{0.25in}}p{\widthof{999}}@{\hspace{0.025in}}p{\linewidth - \widthof{999} - 0.375in}@{}}
                {\raggedright #1.} & {\raggedright #2}\\
        \end{tabular}\par}
